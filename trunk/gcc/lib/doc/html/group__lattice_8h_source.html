238  //check if a subgroup contains the element at prime power order
239  bool Has_prime_pow_el(const cSubgroup<G> grp, const T element)const
240  {
241  std::size_t prime_index = 0;
242  std::size_t elem_order = element.GetOrder();
243  std::size_t prime = boost::math::prime(prime_index);
244  while(prime <= elem_order)
245  {
246  if(grp.Contains(element.GetNthPower(prime)))
247  return true;
248  prime_index++;
249  prime = boost::math::prime(prime_index);
250  }
251  return false;
252  };
