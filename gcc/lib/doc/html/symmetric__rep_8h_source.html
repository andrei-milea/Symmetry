117  std::vector<T> GetElementsDimino()const
118  {
119  assert(!m_GenSet.empty());
120  if(m_GenSet.empty())
121  throw std::logic_error("no generators provided");
122  if(m_GenSet.size() == 1)
123  return GetCyclicSubgroupEl(*m_GenSet.begin());
124 
125  //generate cyclic group of the first generator
126  std::vector<T> elements = GetCyclicSubgroupEl(*m_GenSet.begin());
127 
128  //inductive step
129  for(auto it = (m_GenSet.begin() + 1); it != m_GenSet.end(); it++)
130  {
131  if(find(elements.begin(),elements.end(),*it) == elements.end())
132  {
133  std::size_t prev_order = elements.size() - 1;
134  elements.push_back(*it);
135  for(std::size_t index = 1; index <= prev_order; index++)
136  {
137  if(find(elements.begin(), elements.end(), elements[index] *
138  (*it)) != elements.end())
139  continue;
140  elements.push_back(elements[index] * (*it));
141  }
142  std::size_t rep_pos = prev_order + 1; //cosset rep position
143  while(true)
144  {
145  for(auto it1 = m_GenSet.begin(); it1 != m_GenSet.end(); it1++)
146  {
147  T element = elements[rep_pos] * (*it1);
148  if(find(elements.begin(), elements.end(), element) == elements.end())
149  {
150  elements.push_back(element);
151  for(std::size_t index = 1; index <= prev_order; index++)
152  {
153  if(find(elements.begin(), elements.end(), elements[index] *
154  element) != elements.end())
155  continue;

157  }
158  }
159  }
